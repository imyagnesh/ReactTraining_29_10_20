2. What will be the output for this hoisting example.

var n = 1;

function print() {
    console.log("print():1:",n);
    var n = 2;
    n++;
    console.log("print():2:",n);
}

console.log("inline 1: ", n);
print();
console.log("inline 2: ", n);

Output: inline 1:  1
 print():1: undefined
 print():2: 3
 inline 2:  1

Answ: inside loop global var will not work
